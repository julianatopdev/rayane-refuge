document.addEventListener("DOMContentLoaded", carregarMusicas); async function carregarMusicas(){ const lista = document.getElementById( "listaMusicas" ); try { const resposta = await fetch("../data/musicas.json"); const musicas = await resposta.json(); musicas.forEach( (musica, index)=>{ const card = document.createElement("article"); card.className = "song-card"; card.innerHTML = `
${index === 0 ? "🎵" : "🎵"}

${musica.titulo}

${musica.artista}

${musica.mensagem}

🎧 Ouvir música
`; lista.appendChild(card); } ); } catch(error){ lista.innerHTML = `

Não foi possível carregar nossas músicas 💜

`; console.error(error); } }
